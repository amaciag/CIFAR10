# Load pickle file
def unpickle(file):
    import pickle
    with open(file, 'rb') as fo:
        dict = pickle.load(fo, encoding='bytes')
    return dict

# Reshape the numpy array to (32,32,3) format
def reshaping(data):
    reshape=[]
    for i in range(len(data)):
        a=data[i].reshape((32,32,3))
        reshape.append(a)
    reshape=np.array(reshape)
    return reshape

# Centralize data 
def normalize(X_train,X_test):
    mean = np.mean(X_train,axis=(0,1,2,3))
    std = np.std(X_train, axis=(0,1,2,3))
    X_train = (X_train-mean)/(std+1e-7)
    X_test = (X_test-mean)/(std+1e-7)
    return X_train, X_test

# Convert from categorical array of 1 and 0 to one dimensional array
def convert(d):
    dlist=[int(np.where(d[i]==1)[0][0]) for i in range(len(d))]
    return dlist
